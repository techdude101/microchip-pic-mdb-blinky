# define the MCU
Device PIC16F18076

# set the debugger (hardware tool)
hwtool pkobnano

# program the MCU
program Blink.X\dist\default\production\Blink.X.production.elf

# run the program on the MCU
run
print counter
echo "Start run for 30500 ms = 30.5 seconds"
wait 30500
halt

# print variable(s)
print counter

# exit
quit
